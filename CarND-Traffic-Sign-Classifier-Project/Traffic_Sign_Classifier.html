def grayscale(image):
    RGB2YCrCb = cv2.cvtColor(image, cv2.COLOR_RGB2YCrCb)
    return np.resize(RGB2YCrCb[:,:,0], (32,32,1))

def normalize(image):
    return image/255

def processImage(images):
    ret = []
    if images[0].shape[2] == 1:
        print("image has only one channel")

        print("normalizing...")
    for image in images:
        normalized = 0
        if image.shape[2] == 3:
            normalized = normalize(grayscale(image))
        else:
            normalized = image
        
        ret.append(normalized)
    return np.asarray(ret, dtype=np.float32)

def initialize_parameters():
    """
    Initializes weight parameters to build a neural network with tensorflow. 
    Returns:
    parameters -- a dictionary of tensors containing W1, W2
    """

    tf.set_random_seed(1)  # so that your "random" numbers match ours

    mu = 0
    sigma = 0.1

    # Layer 1: Convolutional. Input = 32x32x1. Output = 28x28x6.
    W1 = tf.Variable(tf.truncated_normal(shape=(5, 5, 1, 6), mean=mu, stddev=sigma))
    # Layer 2a: Convolutional. Output = 10x10x16.
    W2 = tf.Variable(tf.truncated_normal(shape=(5, 5, 6, 16), mean=mu, stddev=sigma))
    # Layer 2b: Convolutional. Output = 10x10x486.
    W3 = tf.Variable(tf.truncated_normal(shape=(6, 6, 6, 6), mean=mu, stddev=sigma))
    # Layer 3: Fully Connected. Input = 900. Output = 84
    W4 = tf.Variable(tf.truncated_normal(shape=(886, 84), mean=mu, stddev=sigma))
    # Layer 4: Fully Connected. Input = 84. Output = 43.
    W5 = tf.Variable(tf.truncated_normal(shape=(84, 43), mean=mu, stddev=sigma))

    parameters = {"W1": W1,
                  "W2": W2,
                  "W3": W3,
                  "W4": W4,
                  "W5": W5,
                  }

    return parameters


def LeNet(x, parameters):
    # Arguments used for tf.truncated_normal, randomly defines variables for the weights and biases for each layer

    W1 = parameters['W1']
    W2 = parameters['W2']
    W3 = parameters['W3']
    W4 = parameters['W4']
    W5 = parameters['W5']
    
    # Layer 1: Convolutional. Input = 32x32x1. Output = 28x28x6.
    conv1 = tf.nn.conv2d(x, W1, strides=[1, 1, 1, 1], padding='VALID')
    # Activation.
    conv1 = tf.nn.relu(conv1)
    # Pooling. Input = 28x28x6. Output = 14x14x6.
    conv1 = tf.nn.max_pool(conv1, ksize=[1, 2, 2, 1], strides=[1, 2, 2, 1], padding='VALID')

    # Layer 2a: Convolutional. Output = 10x10x16.
    conv2 = tf.nn.conv2d(conv1, W2, strides=[1, 1, 1, 1], padding='VALID')
    # Activation.
    conv2 = tf.nn.relu(conv2)
    # Pooling. Input = 10x10x16. Output = 5x5x16.
    conv2 = tf.nn.max_pool(conv2, ksize=[1, 2, 2, 1], strides=[1, 2, 2, 1], padding='VALID')
    
    # Layer 2b Convolutional. Output = 1x1x200.
    conv3 = tf.nn.conv2d(conv1, W3, strides=[1, 1, 1, 1], padding='VALID') 
    # Activation.
    conv3 = tf.nn.relu(conv3)

    # Flatten. Input = 5x5x16. Output = 400.
    conv2_flat = tf.contrib.layers.flatten(conv2)
    conv3_flat = tf.contrib.layers.flatten(conv3)
    
    # Concat conv2_flat and conv3_flat. Input = 400 + 486. Output = 886
    conv_flat = tf.concat_v2([conv2_flat, conv3_flat], 1)
    
    # Layer 3: Fully Connected. Input = 600. Output = 84.
    fc2 = tf.contrib.layers.fully_connected(conv_flat, 886, activation_fn=tf.nn.relu)

    # Layer 5: Fully Connected. Input = 84. Output = 43.
    logits = tf.contrib.layers.fully_connected(fc2, 43, activation_fn=None)

    return logits

def create_placeholders(n_H0, n_W0, n_C0):
    """
    Creates the placeholders for the tensorflow session.

    Arguments:
    n_H0 -- scalar, height of an input image
    n_W0 -- scalar, width of an input image
    n_C0 -- scalar, number of channels of the input

    Returns:
    X -- placeholder for the data input, of shape [None, n_H0, n_W0, n_C0] and dtype "float"
    Y -- placeholder for the input labels, of shape [None, n_y] and dtype "float"
    """

    X = tf.placeholder(tf.float32, [None, n_H0, n_W0, 1])
    Y = tf.placeholder(tf.int32, [None])

    return X, Y

def compute_cost(Z, one_hot_y):
    """
    Computes the cost

    Arguments:
    Z -- output of forward propagation (output of the last LINEAR unit), of shape (43, number of examples)
    Y -- "true" labels vector placeholder, same shape as Z
    C -- scalar, number of classes

    Returns:
    cost - Tensor of the cost function
    """

    cross_entropy = tf.nn.softmax_cross_entropy_with_logits(labels=one_hot_y, logits=Z)
    loss_operation = tf.reduce_mean(cross_entropy)

    return loss_operation

def evaluate(X_data, y_data):
    num_examples = len(X_data)
    total_accuracy = 0
    sess = tf.get_default_session()
    for offset in range(0, num_examples, BATCH_SIZE):
        batch_x, batch_y = X_data[offset:offset+BATCH_SIZE], y_data[offset:offset+BATCH_SIZE]
        accuracy = sess.run(accuracy_operation, feed_dict={x: batch_x, y: batch_y})
        total_accuracy += (accuracy * len(batch_x))
    return total_accuracy / num_examples

EPOCHS = 20
BATCH_SIZE = 128
rate = 0.001

# Create Placeholders of the correct shape
(n_examples, n_H0, n_W0, n_C0) = X_train.shape
x, y = create_placeholders(n_H0, n_W0, n_C0)
keep_prob = tf.placeholder(tf.float32) # probability to keep units
one_hot_y = tf.one_hot(y, 43)

# Initialize parameters
parameters = initialize_parameters()

# Forward propagation: Build the forward propagation in the tensorflow graph
logits = LeNet(x, parameters)

# Cost function: Add cost function to tensorflow graph
loss_operation = compute_cost(logits, one_hot_y)

# Backpropagation: Define the tensorflow optimizer. Use an AdamOptimizer that minimizes the cost.
training_operation = tf.train.AdamOptimizer(learning_rate = rate).minimize(loss_operation)


correct_prediction = tf.equal(tf.argmax(logits, 1), tf.argmax(one_hot_y, 1))
accuracy_operation = tf.reduce_mean(tf.cast(correct_prediction, tf.float32))
saver = tf.train.Saver()

with tf.Session() as sess:
    sess.run(tf.global_variables_initializer())

    print("Training...")
    print()
    for i in range(EPOCHS):
        X_train, y_train = shuffle(X_train, y_train)
        num_batches = int(n_examples / BATCH_SIZE)
        for offset in range(0, n_examples, BATCH_SIZE):
            end = offset + BATCH_SIZE
            batch_x, batch_y = X_train[offset:end], y_train[offset:end]
            _, batch_cost_train, acc_train = sess.run([training_operation, loss_operation, accuracy_operation], \
                                                feed_dict={x: batch_x, y: batch_y, keep_prob: 0.5})
            batch_cost_train += batch_cost_train / num_batches
            acc_train += acc_train / num_batches

        validation_accuracy = evaluate(X_valid, y_valid)
        print("EPOCH {} ...".format(i + 1))
        print("Cost: %f" % batch_cost_train)
        print("Train Accuracy = {:.3f}".format(acc_train))
        print("Validation Accuracy = {:.3f}".format(validation_accuracy))
        print()

    saver.save(sess, './lenet')
    print("Model saved")

Output Top 5 Softmax Probabilities For Each Image Found on the Web¶

For each of the new images, print out the model's softmax probabilities to show the certainty of the model's predictions (limit the output to the top 5 probabilities for each image). tf.nn.top_k could prove helpful here.

The example below demonstrates how tf.nn.top_k can be used to find the top k predictions for each image.

tf.nn.top_k will return the values and indices (class ids) of the top k predictions. So if k=3, for each sign, it'll return the 3 largest probabilities (out of a possible 43) and the correspoding class ids.

Take this numpy array as an example. The values in the array represent predictions. The array contains softmax probabilities for five candidate images with six possible classes. tf.nn.top_k is used to choose the three classes with the highest probability:
